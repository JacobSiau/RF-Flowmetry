102 int main(int argc, char *argv[])
103 {
104  input_t input;
105 
106  initialize_input(&input);
107 
108  signal(SIGINT, interrupt_handler);
109 
110  if (!acc_driver_hal_init())
111  {
112  return EXIT_FAILURE;
113  }
114 
115  if (parse_options(argc, argv, &input) != ACC_STATUS_SUCCESS) {
116  if (input.file_path != NULL) {
117  acc_os_mem_free(input.file_path);
118  }
119  return EXIT_FAILURE;
120  }
121 
122  acc_hal_t hal = acc_driver_hal_get_implementation();
123 
124  hal.log.log_level = input.log_level;
125 
126  if (!acc_rss_activate_with_hal(&hal)) {
127  return EXIT_FAILURE;
128  }
129 
130  acc_service_status_t service_status;
131 
132  switch (input.service_type) {
133  case POWER_BIN:
134  {
135  acc_service_configuration_t power_bin_configuration = set_up_power_bin(&input);
136 
137  if (power_bin_configuration == NULL) {
138  if (input.file_path != NULL) {
139  acc_os_mem_free(input.file_path);
140  }
141  return EXIT_FAILURE;
142  }
143 
144  service_status = execute_power_bin(power_bin_configuration, input.file_path, input.wait_for_interrupt, input.sweep_count);
145 
146  if (input.file_path != NULL) {
147  acc_os_mem_free(input.file_path);
148  }
149 
150  if (service_status != ACC_SERVICE_STATUS_OK) {
151  printf("execute_power_bin() => (%u) %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
152  return EXIT_FAILURE;
153  }
154 
155  acc_service_power_bins_configuration_destroy(&power_bin_configuration);
156  break;
157  }
158 
159  case ENVELOPE:
160  {
161  acc_service_configuration_t envelope_configuration = set_up_envelope(&input);
162 
163  if (envelope_configuration == NULL) {
164  if (input.file_path != NULL) {
165  acc_os_mem_free(input.file_path);
166  }
167  return EXIT_FAILURE;
168  }
169 
170  service_status = execute_envelope(envelope_configuration, input.file_path, input.wait_for_interrupt, input.sweep_count);
171 
172  if (input.file_path != NULL) {
173  acc_os_mem_free(input.file_path);
174  }
175 
176  if (service_status != ACC_SERVICE_STATUS_OK) {
177  printf("execute_envelope() => (%u) %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
178  return EXIT_FAILURE;
179  }
180 
181  acc_service_envelope_configuration_destroy(&envelope_configuration);
182  break;
183  }
184 
185  case IQ:
186  {
187 
188  acc_service_configuration_t iq_configuration = set_up_iq(&input);
189 
190  if (iq_configuration == NULL) {
191  if (input.file_path != NULL) {
192  acc_os_mem_free(input.file_path);
193  }
194  }
195 
196  service_status = execute_iq(iq_configuration, input.file_path, input.wait_for_interrupt, input.sweep_count);
197 
198  if (input.file_path != NULL) {
199  acc_os_mem_free(input.file_path);
200  }
201 
202  if (service_status != ACC_SERVICE_STATUS_OK) {
203  printf("execute_iq() => (%u) %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
204  return EXIT_FAILURE;
205  }
206 
207  acc_service_iq_configuration_destroy(&iq_configuration);
208  break;
209  }
210  default:
211  {
212  printf("Invalid service_type %d\n", input.service_type);
213  return EXIT_FAILURE;
214  }
215  }
216 
217  acc_rss_deactivate();
218 
219  return EXIT_SUCCESS;
220 }

251 bool parse_options(int argc, char *argv[], input_t *input)
252 {
253  static struct option long_options[] =
254  {
255  {"service-type", required_argument, 0, 't'},
256  {"sweep-count", required_argument, 0, 'c'},
257  {"range-start", required_argument, 0, 'b'},
258  {"range-end", required_argument, 0, 'e'},
259  {"frequency", required_argument, 0, 'f'},
260  {"gain", required_argument, 0, 'g'},
261  {"number-of-bins", required_argument, 0, 'n'},
262  {"out", required_argument, 0, 'o'},
263  {"profile", required_argument, 0, 'p'},
264  {"running-avg-factor", required_argument, 0, 'r'},
265  {"sensor", required_argument, 0, 's'},
266  {"verbose", no_argument, 0, 'v'},
267  {"help", no_argument, 0, 'h'},
268  {NULL, 0, NULL, 0}
269  };
270 
271  int16_t character_code;
272  int32_t option_index = 0;
273 
274  while ((character_code = getopt_long(argc, argv, "t:c:b:e:f:g:n:o:p:r:s:vh?", long_options, &option_index)) != -1) {
275  switch (character_code) {
276  case 't':
277  {
278  switch (atoi(optarg)) {
279  case 0:
280  {
281  input->service_type = POWER_BIN;
282  break;
283  }
284  case 1:
285  {
286  input->service_type = ENVELOPE;
287  break;
288  }
289  case 2:
290  {
291  input->service_type = IQ;
292  break;
293  }
294  default:
295  printf("Invalid service type.\n");
296  print_usage();
297  return ACC_STATUS_FAILURE;
298  }
299  break;
300  }
301  case 'c':
302  {
303  input->sweep_count = atoi(optarg);
304  input->wait_for_interrupt = false;
305  break;
306  }
307  case 'b':
308  {
309  char *next;
310  input->start_m = strtof(optarg, &next);
311  break;
312  }
313  case 'e':
314  {
315  char *next;
316  input->end_m = strtof(optarg, &next);
317  break;
318  }
319  case 'f':
320  {
321  char *next;
322  float f = strtof(optarg, &next);
323  if (f > 0 && f < 100000) {
324  input->frequency = f;
325  }
326  else {
327  printf("Frequency out of range.\n");
328  print_usage();
329  exit(EXIT_FAILURE);
330  }
331  break;
332  }
333  case 'g':
334  {
335  char *next;
336  float g = strtof(optarg, &next);
337  if (g >= 0 && g <= 1) {
338  input->gain = g;
339  }
340  else {
341  printf("Gain out of range.\n");
342  print_usage();
343  exit(EXIT_FAILURE);
344  }
345  break;
346  }
347  case 'n':
348  {
349  int n = atoi(optarg);
350  if (n > 0 && n <= 32) {
351  input->n_bins = n;
352  }
353  else {
354  printf("Number of bins out of range.\n");
355  print_usage();
356  exit(EXIT_FAILURE);
357  }
358  break;
359  }
360  case 'p':
361  {
362  switch (atoi(optarg)) {
363  case 0:
364  {
365  input->profile = ACC_SERVICE_ENVELOPE_PROFILE_MAXIMIZE_SNR;
366  break;
367  }
368  case 1:
369  {
370  input->profile = ACC_SERVICE_ENVELOPE_PROFILE_MAXIMIZE_DEPTH_RESOLUTION;
371  break;
372  }
373  default:
374  printf("Invalid profile.\n");
375  print_usage();
376  exit(EXIT_FAILURE);
377  }
378  break;
379  }
380  case 'o':
381  {
382  input->file_path = acc_os_mem_alloc(sizeof(char) * (strlen(optarg) + 1));
383  if (input->file_path == NULL) {
384  printf("Failed allocating memory\n");
385  return ACC_STATUS_FAILURE;
386  }
387  snprintf(input->file_path, strlen(optarg) + 1, "%s", optarg);
388  break;
389  }
390  case 'r':
391  {
392  char *next;
393  float r = strtof(optarg, &next);
394  if (r >= 0 && r <= 1) {
395  input->running_avg = r;
396  }
397  else {
398  printf("Running average factor out of range.\n");
399  print_usage();
400  exit(EXIT_FAILURE);
401  }
402  break;
403  }
404  case 's':
405  {
406  int s = atoi(optarg);
407  if (s > 0 && s <= 4) {
408  input->sensor = s;
409  }
410  else {
411  printf("Sensor id out of range.\n");
412  print_usage();
413  exit(EXIT_FAILURE);
414  }
415  break;
416  }
417  case 'v':
418  {
419  input->log_level = ACC_LOG_LEVEL_VERBOSE;
420  break;
421  }
422  case 'h':
423  case '?':
424  {
425  print_usage();
426  return ACC_STATUS_FAILURE;
427  }
428  }
429  }
430 
431  if (input->service_type == INVALID_SERVICE) {
432  printf("Missing option service type.\n");
433  print_usage();
434  return ACC_STATUS_FAILURE;
435  }
436 
437  return ACC_STATUS_SUCCESS;
438 }

475 acc_service_status_t execute_power_bin(acc_service_configuration_t power_bin_configuration, char *file_path, bool wait_for_interrupt, uint16_t sweep_count)
476 {
477  acc_service_handle_t handle = acc_service_create(power_bin_configuration);
478 
479  if (handle == NULL) {
480  printf("acc_service_create failed\n");
481  return ACC_SERVICE_STATUS_FAILURE_UNSPECIFIED;
482  }
483 
484  acc_service_power_bins_metadata_t power_bins_metadata;
485  acc_service_power_bins_get_metadata(handle, &power_bins_metadata);
486 
487  float power_bins_data[power_bins_metadata.actual_bin_count];
488 
489  acc_service_power_bins_result_info_t result_info;
490  acc_service_status_t service_status = acc_service_activate(handle);
491 
492  if (service_status == ACC_SERVICE_STATUS_OK) {
493  FILE *file = stdout;
494 
495  if (file_path != NULL) {
496  file = fopen(file_path, "w");
497 
498  if(file == NULL) {
499  printf("opening file failed\n");
500  return ACC_SERVICE_STATUS_FAILURE_UNSPECIFIED;
501  }
502  }
503 
504  uint16_t sweeps = 0;
505 
506  while ((wait_for_interrupt && interrupted == 0) || sweeps < sweep_count) {
507  service_status = acc_service_power_bins_get_next(handle, power_bins_data, power_bins_metadata.actual_bin_count, &result_info);
508 
509  if (service_status == ACC_SERVICE_STATUS_OK) {
510  for (uint_fast16_t index = 0; index < power_bins_metadata.actual_bin_count; index++) {
511  fprintf(file, "%u\t", (unsigned int)(power_bins_data[index] + 0.5f));
512  }
513  fprintf(file, "\n");
514 
515  if (file_path == NULL) {
516  fflush(stdout);
517  }
518  }
519  else {
520  printf("Power bin data not properly retrieved\n");
521  fflush(stdout);
522  return ACC_SERVICE_STATUS_FAILURE_UNSPECIFIED;
523  }
524 
525  if (!wait_for_interrupt) {
526  sweeps++;
527  }
528  }
529 
530  if (file_path != NULL) {
531  fclose(file);
532  }
533 
534  service_status = acc_service_deactivate(handle);
535  }
536  else {
537  printf("acc_service_activate() (%u) => %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
538  }
539 
540  acc_service_destroy(&handle);
541 
542  return service_status;
543 }

738  }
739

741  sweeps++;
742  }
743  }
744

747  }
748 
749  service_status = acc_service_deactivate(handle);
750  }
751  else {
