147 /**
148  * @brief Get running average factor
149  *
150  * The running average factor is the factor of which the most recent sweep is weighed against previous sweeps.
151  * Valid range is between 0.0 and 1.0 where 0.0 means that no history is weighed in, i.e filtering is effectively disabled.
152  * A factor of 1.0 means that the most recent sweep has no effect on the result,
153  * which will result in that the first sweep is forever received as the result.
154  * The filtering is coherent and is done on complex valued IQ data before conversion to envelope data.
155  *
156  * @param[in] service_configuration The configuration to get the running average factor for
157  * @return Running average factor
158  */
159 extern float acc_service_envelope_running_average_factor_get(acc_service_configuration_t service_configuration);
