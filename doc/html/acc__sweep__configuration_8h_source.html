21 /**
22  * @brief Power save mode
23  *
24  * Each power save mode corresponds to a how much of the sensor hardware is shutdown
25  * between sweeps. Mode A means that the whole sensor is shutdown between sweeps while
26  * mode D means that the sensor is in its active state all the time.
27  * For each power save mode there will be a limit in the achievable update rate. Mode A
28  * will have the lowest update rate limit but also consumes the least amount of power for
29  * low update rates.
30  * The update rate limits also depend on integration and range settings so for each
31  * scenario it is up to the user to find the best possible compromise between update rate
32  * and power consumption.
33  */
34 typedef enum
35 {
36  ACC_SWEEP_CONFIGURATION_POWER_SAVE_MODE_A,
37  ACC_SWEEP_CONFIGURATION_POWER_SAVE_MODE_B,
38  ACC_SWEEP_CONFIGURATION_POWER_SAVE_MODE_C,
39  ACC_SWEEP_CONFIGURATION_POWER_SAVE_MODE_D,
40  ACC_SWEEP_CONFIGURATION_POWER_SAVE_MODE_COUNT
41 } acc_sweep_configuration_power_save_mode_enum_t;
42 typedef uint32_t acc_sweep_configuration_power_save_mode_t;
