72 /**
73  * @brief Get running average factor
74  *
75  * The running average factor is the factor of which the most recent sweep is weighed against previous sweeps.
76  * Valid range is between 0.0 and 1.0 where 0.0 means that no history is weighed in, i.e filtering is effectively disabled.
77  * A factor of 1.0 means that the most recent sweep has no effect on the result,
78  * which will result in that the first sweep is forever received as the result.
79  *
80  * @param[in] service_configuration The configuration to set the running average factor for
81  * @return Running average factor
82  */
83 extern float acc_service_iq_running_average_factor_get(acc_service_configuration_t service_configuration);
