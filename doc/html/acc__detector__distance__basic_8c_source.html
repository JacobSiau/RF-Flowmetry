43 acc_detector_distance_basic_handle_t *acc_detector_distance_basic_create(acc_sensor_id_t sensor, float range_start, float range_length)
44 {
45  acc_service_configuration_t envelope_configuration = acc_service_envelope_configuration_create();
46 
47  if (envelope_configuration == NULL)
48  {
49  ACC_LOG_ERROR("Distance basic detector not possible to allocate");
50  return NULL;
51  }
52 
53  acc_service_envelope_profile_set(envelope_configuration, ACC_SERVICE_ENVELOPE_PROFILE_MAXIMIZE_SNR);
54 
55  acc_sweep_configuration_t sweep_configuration = acc_service_get_sweep_configuration(envelope_configuration);
56 
57  acc_sweep_configuration_repetition_mode_max_frequency_set(sweep_configuration);
58  acc_sweep_configuration_requested_range_set(sweep_configuration, range_start, range_length);
59  acc_sweep_configuration_sensor_set(sweep_configuration, sensor);
60 
61  acc_detector_distance_basic_handle_internal_t *detector_handle = acc_os_mem_alloc(sizeof(*detector_handle));
62 
63  if (detector_handle != NULL)
64  {
65  detector_handle->magic_number = MAGIC_NUMBER;
66  detector_handle->envelope_handle = acc_service_create(envelope_configuration);
67 
68  if (detector_handle->envelope_handle != NULL)
69  {
70  acc_service_envelope_metadata_t envelope_metadata;
71  acc_service_envelope_get_metadata(detector_handle->envelope_handle, &envelope_metadata);
72 
73  detector_handle->absolute_offset = envelope_metadata.free_space_absolute_offset;
74  detector_handle->envelope_data_length = envelope_metadata.data_length;
75  detector_handle->envelope_data = acc_os_mem_alloc(sizeof(*detector_handle->envelope_data) * detector_handle->envelope_data_length);
76  detector_handle->distance_offset = envelope_metadata.actual_start_m;
77  detector_handle->distance_slope = envelope_metadata.actual_length_m / (float)(envelope_metadata.data_length - 1);
78 
79  if (detector_handle->envelope_data != NULL)
80  {
81  acc_service_activate(detector_handle->envelope_handle);
82  }
83  else
84  {
85  acc_service_destroy(&detector_handle->envelope_handle);
86  acc_os_mem_free(detector_handle);
87  detector_handle = NULL;
88  ACC_LOG_ERROR("Distance basic detector not possible to allocate");
89  }
90  }
91  else
92  {
93  acc_os_mem_free(detector_handle);
94  detector_handle = NULL;
95  ACC_LOG_ERROR("Distance basic detector not possible to create");
96  }
97  }
98  else
99  {
100  detector_handle = NULL;
101  ACC_LOG_ERROR("Distance basic detector not possible to allocate");
102  }
103 
104  acc_service_envelope_configuration_destroy(&envelope_configuration);
105 
106  return (acc_detector_distance_basic_handle_t*)detector_handle;
107 }

126 acc_detector_distance_basic_reflection_t acc_detector_distance_basic_get_reflection(acc_detector_distance_basic_handle_t *handle)
127 {
128  acc_detector_distance_basic_reflection_t reflection;
129  reflection.distance = 0.0f;
130  reflection.amplitude = 0.0f;
131 
132  if (handle_valid(handle))
133  {
134  uint16_t *envelope_data = handle->envelope_data;
135  uint16_t envelope_data_length = handle->envelope_data_length;
136 
137  acc_service_envelope_result_info_t result_info;
138  acc_service_envelope_get_next(handle->envelope_handle, envelope_data, envelope_data_length, &result_info);
139 
140  uint16_t reflection_index = get_reflection_index(envelope_data, envelope_data_length);
141 
142  reflection.amplitude = envelope_data[reflection_index];
143  reflection.distance = INDEX_TO_DISTANCE_CONVERSION(reflection_index, handle->distance_slope, handle->distance_offset);
144  }
145 
146  return reflection;

167 uint16_t get_reflection_index(uint16_t *data, uint16_t data_length)
168 {
169  uint16_t reflection_amplitude = 0;
170  uint16_t reflection_index = 0;
171  for (uint16_t i = 0; i < data_length; i++)
172  {
173  if (data[i] > reflection_amplitude)
174  {
175  reflection_amplitude = data[i];
176  reflection_index = i;
177  }
178  }
179 
180  return reflection_index;
